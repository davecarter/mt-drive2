Code (HTML)

<form action="" class="mt-BasicForm">

  <!-- Use <fieldset> to group form controls -->
  <fieldset>

    <!-- Use <legend> to title a <fieldset> -->
    <legend>User Login</legend>

    <!-- Use the class "mt-BasicForm-group" to group a control and it's label -->
    <div class="mt-BasicForm-group">
      <label for="input-text">Input text</label>
      <input id="input-text" type="text" placeholder="Enter your name">
    </div>

    <div class="mt-BasicForm-group">
      <label for="input-mail">Input mail</label>
      <input id="input-mail" type="mail" placeholder="Enter your mail (example@mail.com)">
    </div>

    <div class="mt-BasicForm-group">
      <label for="input-password">Input password</label>
      <input id="input-password" type="password" placeholder="Enter password (min. 8 characters)">
    </div>
  </fieldset>

  <fieldset>
    <legend>User Login</legend>

    <div class="mt-BasicForm-group">
      <label for="select">Select</label>
      <select id="select">
        <option value="">Option 1</option>
        <option value="">Option 2</option>
        <option value="">Option 3</option>
        <option value="">Option 4</option>
      </select>
    </div>

    <div class="mt-BasicForm-group">
      <label for="textarea">Textarea</label>
      <textarea id="textarea" placeholder="Write some text here..."></textarea>
    </div>
  </fieldset>

  <fieldset>
    <div class="mt-BasicForm-group">

      <!-- Put input types checkbox and radio inside their labels -->
      <label for="input-checkbox">
        <input id="input-checkbox" type="checkbox">
        Checkbox
      </label>

    </div>

    <div class="mt-BasicForm-group">
      <label for="input-radio">
        <input id="input-radio" type="radio">
        Radio
      </label>
    </div>
  </fieldset>
</form>
